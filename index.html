import matplotlib.pyplot as plt
import numpy as np
import random



def tree(x, y, angle, depth, branch_length):
    if depth == 0:
        return
    
    
    x2 = x + int(np.cos(np.radians(angle)) * branch_length)
    y2 = y + int(np.sin(np.radians(angle)) * branch_length)
    plt.plot([x, x2], [y, y2], color='k', linewidth=2.0)

    tree(x2, y2, angle - 60/random.uniform(2, 5), depth - 1, branch_length/random.uniform(1.1, 2))
    tree(x2, y2, angle + 60/random.uniform(2, 5), depth - 1, branch_length/random.uniform(1.1, 2))
    tree(x2, y2, angle/random.uniform(2, 5), depth - 1, branch_length//random.uniform(1.1, 2))

plt.figure(figsize=(8,8))
plt.axis('equal')
plt.axis('off')

tree(300, 800, 90, 10, 100)

plt.show()
